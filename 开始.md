# 验证 Java 环境
java -version
# 验证 Allure 版本
allure --version
# 验证 pytest-allure 插件
pip list | findstr allure-pytest  # Windows
# pip list | grep allure-pytest  # macOS/Linux


# 执行 tests 目录下所有测试用例，生成 Allure 原始报告数据（JSON/XML 格式）
# -v：显示用例执行详情；-s：输出用例中的 print 日志
# --alluredir：指定原始数据输出目录（Report/allure-results）
# --clean-alluredir：执行前清空原始数据目录，避免历史数据干扰
pytest -vs tests/test_sd_ui.py --alluredir=Report/allure-results --clean-alluredir
# 基于原始 JSON 数据生成 Allure HTML 报告
# generate：生成 HTML 报告的核心命令
# -c/--clean：清空目标报告目录（Report/allure-reports）后生成新报告（等价于 --clean）
# -o：指定 HTML 报告输出目录
allure generate -c Report/allure-results -o Report/allure-reports
# 启动临时本地服务，自动在默认浏览器打开 Allure 报告
# serve：启动 web 服务（无需手动打开 HTML 文件，兼容性更好）
# Report/allure-results：指定原始数据目录（serve 会基于此动态生成报告）
allure serve Report/allure-results


# 把项目中所ALL变更（新增/修改/删除）加入暂存区（全量添加）
git add .  
# 把项目中指定文件添加到暂存区
git add .XX


# 清空暂存区所有文件，本地代码保持不变（仅撤销git add）
git reset
# 清空暂存区 + 回滚本地代码到上次提交
git reset --hard HEAD	
# 格式：git reset HEAD <文件/目录路径>
git reset HEAD .gitignore  # 仅清空暂存区中的.gitignore文件
git reset HEAD scripts/    # 仅清空暂存区中的scripts目录

# 提交暂存代码到我的本地
git commit -m "修复失败"
# 提交本地到远程
git push origin master

# 切换分支
git checkout master
